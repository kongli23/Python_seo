//创建表, id主键自增,keywords关键词,title文章标题,content文章内容
create table article(
	id int primary key auto_increment,
	keywords varchar(120),
	title varchar(255),
	content text
)

//清空数据表
delete from article;
//并重置数据表中自增id
alter table article auto_increment=1

//修改表,设定内容字段不允许重复
alter table article add unique(title);
select * from article

//插入数据,如果有重复则直接忽略
insert ignore into article(keywords,title,content) values('seo','seo是什么意思标题','seo是什么意思文章')

select * from article where keywords='名人名言故事'
